     @Override
     public double priceToAggregate(List<Item> cart) {
         boolean hasAnElectronicDevice = cart.stream().anyMatch(it -> it.getType() == ItemType.ELECTRONIC);
 
         if(hasAnElectronicDevice)
             return 7.50;

     }
